<script>
// 🔥 MULTI API BACKUP
const APIs = [
  (url) => `https://api.tiklydown.eu.org/api/download?url=${url}`,
  (url) => `https://tikwm.com/api/?url=${url}`
];

// 🔄 FETCH DENGAN RETRY
async function fetchWithBackup(url) {
  for (let api of APIs) {
    try {
      let res = await fetch(api(encodeURIComponent(url)));
      let data = await res.json();

      // format beda tiap API
      if (data.video) return {video:data.video, music:data.music};
      if (data.data) return {
        video: data.data.play,
        music: data.data.music
      };

    } catch(e) {}
  }
  throw "Semua server gagal";
}

// 🎬 VIDEO
async function getVideo() {
  const url = document.getElementById("url").value;
  const result = document.getElementById("result");

  if (!url.includes("tiktok")) {
    result.innerHTML = "❌ Link tidak valid!";
    return;
  }

  result.innerHTML = "<div class='loading'>Loading...</div>";

  try {
    let data = await fetchWithBackup(url);

    result.innerHTML = `
      <video controls src="${data.video}"></video>
      <a href="${data.video}" download>
        <button>⬇ Download</button>
      </a>
    `;

  } catch {
    result.innerHTML = "🚫 Server sibuk, coba lagi!";
  }
}

// 🎵 MP3
async function getMP3() {
  const url = document.getElementById("url").value;
  const result = document.getElementById("result");

  result.innerHTML = "<div class='loading'>Converting...</div>";

  try {
    let data = await fetchWithBackup(url);

    result.innerHTML = `
      <audio controls src="${data.music}"></audio>
      <a href="${data.music}" download>
        <button>⬇ Download MP3</button>
      </a>
    `;

  } catch {
    result.innerHTML = "🚫 Gagal ambil audio!";
  }
}
</script>
